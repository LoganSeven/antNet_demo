cmake_minimum_required(VERSION 3.10)

# ------------------ Project setup ---------------------------------
project(AntNetBackend VERSION 0.1.0 LANGUAGES C)
set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Default to Debug if not specified
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE "Debug" CACHE STRING "Build type (Debug or Release)" FORCE)
endif()

message(STATUS "‚õèÔ∏è  Build type: ${CMAKE_BUILD_TYPE}")
message(STATUS "üíª Target platform: ${CMAKE_SYSTEM_NAME}")

# ------------------ Compiler warnings/options -----------------------------
if(MSVC)
    add_compile_options(/W4)
else()
    add_compile_options(-Wall -Wextra -pedantic)
endif()

# ------------------ Include directories ---------------------------
# Ensure this covers error_codes.h, random_algo.h, etc.
include_directories(
    ${CMAKE_SOURCE_DIR}/include
    ${CMAKE_SOURCE_DIR}/include/algo
    ${CMAKE_SOURCE_DIR}/include/algo/cpu
    ${CMAKE_SOURCE_DIR}/include/rendering
    ${CMAKE_SOURCE_DIR}/include/types
    ${CMAKE_SOURCE_DIR}/third_party
)

# ------------------ Source files ----------------------------------
set(SOURCE_FILES
    src/c/hop_map_manager.c
    src/c/backend.c
    src/c/backend_topology.c
    src/c/random_algo.c
    src/c/cpu_brute_force.c
    src/c/cpu_ACOv1.c
    src/c/config_manager.c
    src/c/heatmap_renderer.c
    src/c/heatmap_renderer_async.c   # <--- Added async renderer
    third_party/ini.c
)

# ------------------ OpenGL ES / EGL -------------------------------
find_library(EGL_LIB EGL REQUIRED)
find_library(GLESv2_LIB GLESv2 REQUIRED)

# ------------------ Build shared library --------------------------
add_library(antnet_backend SHARED ${SOURCE_FILES})

# ------------------ Linking ---------------------------------------
if(UNIX AND NOT APPLE)
    find_package(Threads REQUIRED)
    target_link_libraries(antnet_backend
        PRIVATE
            m
            Threads::Threads
            ${EGL_LIB}
            ${GLESv2_LIB}
    )
endif()

# ------------------ Install targets (optional) --------------------
install(TARGETS antnet_backend
        LIBRARY DESTINATION lib
        ARCHIVE DESTINATION lib
        RUNTIME DESTINATION bin)

install(DIRECTORY include/ DESTINATION include)

# ------------------ Final Summary -----------------------------------
message(STATUS "‚úÖ AntNet backend build setup complete.")
message(STATUS "üì¶ Output: shared library 'antnet_backend' for CFFI loading.")
