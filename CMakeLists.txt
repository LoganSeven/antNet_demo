cmake_minimum_required(VERSION 3.10)

# ------------------ Project setup ---------------------------------
project(AntNetBackend VERSION 0.1.0 LANGUAGES C)
set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Default to Debug if not specified
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE "Debug" CACHE STRING "Build type (Debug or Release)" FORCE)
endif()

message(STATUS "‚õèÔ∏è  Build type: ${CMAKE_BUILD_TYPE}")

# ------------------ Compiler warnings/options -----------------------------
if(MSVC)
    add_compile_options(/W4)
else()
    add_compile_options(-Wall -Wextra -pedantic)
endif()

# ------------------ Include directories ---------------------------
include_directories(
    ${CMAKE_SOURCE_DIR}/include
)

# ------------------ Source files ----------------------------------
set(SOURCE_FILES
    src/c/backend.c
    # Add other .c files here if needed
)

# ------------------ Build shared library --------------------------
add_library(antnet_backend SHARED ${SOURCE_FILES})

# ------------------ Linking ---------------------------------------
# (Add if you later link to math or pthread explicitly)
if(UNIX AND NOT APPLE)
    target_link_libraries(antnet_backend PRIVATE m)
endif()

# ------------------ Install targets (optional) --------------------
install(TARGETS antnet_backend
        LIBRARY DESTINATION lib
        ARCHIVE DESTINATION lib
        RUNTIME DESTINATION bin)

install(DIRECTORY include/ DESTINATION include)
